export const DocIconAuto = ({ title }) => {
  const [isOpen, setIsOpen] = React.useState(false);
  const [placement, setPlacement] = React.useState('top');
  const [bubbleStyle, setBubbleStyle] = React.useState(null);
  const iconRef = React.useRef(null);
  const bubbleRef = React.useRef(null);

  const recompute = React.useCallback(() => {
    const icon = iconRef.current;
    const bubble = bubbleRef.current;
    if (!icon || !bubble) return;

    const iconRect = icon.getBoundingClientRect();
    const bubbleRect = bubble.getBoundingClientRect();

    const gap = 10;
    const viewportPadding = 12;
    const sidebar = document.querySelector('aside.theme-doc-sidebar-container');
    const sidebarRect = sidebar?.getBoundingClientRect();
    const leftSafe = Math.max(
      viewportPadding,
      sidebarRect ? sidebarRect.right + viewportPadding : viewportPadding
    );

    const canPlaceTop = iconRect.top >= bubbleRect.height + gap + viewportPadding;
    const nextPlacement = canPlaceTop ? 'top' : 'bottom';
    setPlacement(nextPlacement);

    const desiredLeft = iconRect.left + iconRect.width / 2 - bubbleRect.width / 2;
    const minLeft = leftSafe;
    const maxLeft = window.innerWidth - viewportPadding - bubbleRect.width;
    const left = Math.min(Math.max(desiredLeft, minLeft), maxLeft);

    const top =
      nextPlacement === 'top'
        ? iconRect.top - gap - bubbleRect.height
        : iconRect.bottom + gap;

    const iconCenterX = iconRect.left + iconRect.width / 2;
    const arrowX = Math.min(Math.max(iconCenterX - left, 14), bubbleRect.width - 14);

    setBubbleStyle({
      left,
      top,
      ['--aqara-doc-tooltip-arrow-x']: `${arrowX}px`,
    });
  }, []);

  const open = React.useCallback(() => {
    setIsOpen(true);
  }, []);

  const close = React.useCallback(() => {
    setIsOpen(false);
    setPlacement('top');
    setBubbleStyle(null);
  }, []);

  React.useLayoutEffect(() => {
    if (!isOpen) return;
    recompute();
  }, [isOpen, title, recompute]);

  React.useEffect(() => {
    if (!isOpen) return;
    const handler = () => recompute();
    window.addEventListener('scroll', handler, true);
    window.addEventListener('resize', handler);
    return () => {
      window.removeEventListener('scroll', handler, true);
      window.removeEventListener('resize', handler);
    };
  }, [isOpen, recompute]);

  return (
    <span className="aqara-doc-tooltip">
      <span
        className="aqara-doc-tooltip__icon"
        ref={iconRef}
        tabIndex={0}
        role="button"
        aria-label="查看说明"
        onMouseEnter={open}
        onMouseLeave={close}
        onFocus={open}
        onBlur={close}
      >
        <svg
          xmlns="http://www.w3.org/2000/svg"
          width="14"
          height="14"
          viewBox="0 0 24 24"
          fill="none"
          stroke="currentColor"
          strokeWidth="2"
          strokeLinecap="round"
          strokeLinejoin="round"
          aria-hidden="true"
          focusable="false"
        >
          <path d="M14.5 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V7.5L14.5 2z" />
          <polyline points="14 2 14 8 20 8" />
          <line x1="16" y1="13" x2="8" y2="13" />
          <line x1="16" y1="17" x2="8" y2="17" />
          <line x1="10" y1="9" x2="8" y2="9" />
        </svg>
      </span>
      {isOpen &&
        ReactDOM.createPortal(
          <span
            ref={bubbleRef}
            className={[
              'aqara-doc-tooltip__bubble',
              'aqara-doc-tooltip__bubble--portal',
              placement === 'bottom' ? 'aqara-doc-tooltip__bubble--bottom' : '',
            ]
              .filter(Boolean)
              .join(' ')}
            style={bubbleStyle || undefined}
            role="tooltip"
          >
            {title}
          </span>,
          document.body
        )}
    </span>
  );
};

选型建议：
- M300 / GX300M：作为 Aqara Studio 官方标准主机，预装 Aqara Studio OS 并内置中枢 Hub，实现开箱即用的本地化部署与运行；凭借稳定的边缘计算能力，完美适配住宅及小型商业项目的智能化需求。
- AX100M / AX200M：作为 Aqara Studio 子网关，需接入 M300 / GX300M 等搭载 Aqara Studio 的官方标准主机，主要用于扩展 Zigbee 与 Thread 无线网络的覆盖范围和设备接入容量；适用于需分布式部署、多区域信号延伸或协议桥接（如红外控制）的场景。
