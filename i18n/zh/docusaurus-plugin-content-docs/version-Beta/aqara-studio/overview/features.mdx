export const DocIconAuto = ({ title }) => {
  const [isOpen, setIsOpen] = React.useState(false);
  const [placement, setPlacement] = React.useState('top');
  const rootRef = React.useRef(null);
  const bubbleRef = React.useRef(null);

  const recomputePlacement = React.useCallback(() => {
    const root = rootRef.current;
    const bubble = bubbleRef.current;
    if (!root || !bubble) return;

    const rootRect = root.getBoundingClientRect();
    const bubbleRect = bubble.getBoundingClientRect();
    const requiredTop = bubbleRect.height + 12;

    setPlacement(rootRect.top < requiredTop ? 'bottom' : 'top');
  }, []);

  const open = React.useCallback(() => {
    setIsOpen(true);
    requestAnimationFrame(recomputePlacement);
  }, [recomputePlacement]);

  const close = React.useCallback(() => {
    setIsOpen(false);
    setPlacement('top');
  }, []);

  return (
    <span
      ref={rootRef}
      className={[
        'aqara-doc-tooltip',
        isOpen ? 'aqara-doc-tooltip--open' : '',
        placement === 'bottom' ? 'aqara-doc-tooltip--bottom' : '',
      ]
        .filter(Boolean)
        .join(' ')}
    >
      <span
        className="aqara-doc-tooltip__icon"
        tabIndex={0}
        role="button"
        aria-label="查看说明"
        onMouseEnter={open}
        onMouseLeave={close}
        onFocus={open}
        onBlur={close}
      >
        <svg
          xmlns="http://www.w3.org/2000/svg"
          width="14"
          height="14"
          viewBox="0 0 24 24"
          fill="none"
          stroke="currentColor"
          strokeWidth="2"
          strokeLinecap="round"
          strokeLinejoin="round"
          aria-hidden="true"
          focusable="false"
        >
          <path d="M14.5 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V7.5L14.5 2z" />
          <polyline points="14 2 14 8 20 8" />
          <line x1="16" y1="13" x2="8" y2="13" />
          <line x1="16" y1="17" x2="8" y2="17" />
          <line x1="10" y1="9" x2="8" y2="9" />
        </svg>
      </span>
      <span
        ref={bubbleRef}
        className="aqara-doc-tooltip__bubble"
        role="tooltip"
      >
        {title}
      </span>
    </span>
  );
};
